> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the next-stage QA API on top of the existing RAG minimum loop, including intent routing, query rewrite, RAG/general-LLM routing, answer generation, refusal thresholds, and QA persistence.

**Architecture:** Add a thin FastAPI QA endpoint that delegates to a service layer. The service layer first applies deterministic guardrails, then uses an LLM for intent classification and query rewrite when needed, then routes to RAG, general LLM, or refusal. RAG answers must use retrieved evidence and persist references; general LLM answers return empty references; low-confidence knowledge-base questions are refused and recorded in `qa_unanswered`.

**Tech Stack:** FastAPI, Pydantic, SQLAlchemy 2.0, Alembic, PostgreSQL 16, pgvector, SiliconFlow chat API, SiliconFlow embedding API, SiliconFlow rerank API, pytest, httpx.

## Confirmed Product Rules

Use these rules as fixed requirements:

1. `retrieval_final_top_k` stays at 5 for evidence retrieval.
2. Final answer references should normally include Top 3-5 evidence chunks for RAG answers.
3. Initial rerank refusal threshold is configurable and should default to `0.2`.
4. Medium-confidence RAG answers should be cautious instead of immediately refused.
5. Answers may use model-native knowledge only when the intent is `general_explanation`; in that case `references` must be empty and the response must clearly report that the project knowledge base was not used.

## Intent Routing Requirements

Intent routing must use three layers:

1. **Hard rules**
   - Empty, all-punctuation, or too-short input: `invalid_input`.
   - Obvious realtime/external questions such as weather, latest news, stock price, exchange rate, current time: `realtime_external`.
   - Domain term plus fault/action term: force `knowledge_base_qa`.

2. **LLM intent classification + query rewrite**
   - Used for non-obvious questions.
   - Must output structured JSON.
   - Must not answer the user question.
   - Must preserve original meaning, device names, fault codes, model numbers, English abbreviations, and technical terms.

3. **Post-validation**
   - If the LLM says `general_explanation` but the question contains domain terms plus fault/action terms, override to `knowledge_base_qa`.
   - If `search_query` is empty for `knowledge_base_qa`, fall back to `normalized_question`.
   - If LLM JSON parsing fails, use rule-based fallback.

## Answer Type Requirements

Support these API-level answer types:

- `rag`: evidence-based project knowledge-base answer.
- `general_llm`: model-native general explanation answer with empty references.
- `refused`: no reliable answer.
- `none`: internal/default state before final decision.

The existing database enum currently contains `faq`, `rag`, `fallback`, and `none`. Add `general_llm` and `refused` to the enum through an Alembic migration and update the SQLAlchemy enum.

```python
class SiliconFlowChatClient:
    def __init__(
        self,
        client: httpx.AsyncClient,
        api_key: str,
        model: str,
    ) -> None:
        self._client = client
        self._api_key = api_key
        self._model = model

    async def chat(
        self,
        messages: list[dict[str, str]],
        temperature: float = 0.1,
    ) -> str:
        response = await self._client.post(
            "/chat/completions",
            headers=self._headers(),
            json={
                "model": self._model,
                "messages": messages,
                "temperature": temperature,
            },
        )
        response.raise_for_status()
        return response.json()["choices"][0]["message"]["content"]

    def _headers(self) -> dict[str, str]:
        return {"Authorization": f"Bearer {self._api_key}"}
```

from __future__ import annotations

from dataclasses import dataclass
from enum import Enum
import json
import re
from typing import Protocol

from app.services.keyword_index import normalize_query


class ChatClient(Protocol):
    async def chat(
        self,
        messages: list[dict[str, str]],
        temperature: float = 0.1,
    ) -> str:
        ...


class Intent(str, Enum):
    knowledge_base_qa = "knowledge_base_qa"
    general_explanation = "general_explanation"
    out_of_scope = "out_of_scope"
    realtime_external = "realtime_external"
    invalid_input = "invalid_input"


@dataclass(frozen=True)
class QueryUnderstandingResult:
    intent: Intent
    confidence: float
    should_use_knowledge_base: bool
    normalized_question: str
    search_query: str
    refusal_reason: str | None
    reason: str


DOMAIN_TERMS = (
    "逆变器",

    "箱变",

    "线缆",

    "价格",

def apply_intent_hard_rules(
    question: str,
    max_question_chars: int = 500,
) -> QueryUnderstandingResult | None:
    normalized = normalize_query(question)
    if _is_invalid_question(normalized, max_question_chars):
        return QueryUnderstandingResult(
            intent=Intent.invalid_input,
            confidence=1.0,
            should_use_knowledge_base=False,
            normalized_question=normalized,
            search_query="",
            refusal_reason="invalid_input",
            reason="问题为空、过长或缺少可识别文字",
        )

    if _contains_any(normalized, REALTIME_TERMS) and not _is_domain_fault_question(normalized):
        return QueryUnderstandingResult(
            intent=Intent.realtime_external,
            confidence=1.0,
            should_use_knowledge_base=False,
            normalized_question=normalized,
            search_query="",
            refusal_reason="realtime_external",
            reason="问题需要实时外部信息，当前系统未接入外部实时数据源",
        )

    if _is_domain_fault_question(normalized):
        return QueryUnderstandingResult(
            intent=Intent.knowledge_base_qa,
            confidence=1.0,
            should_use_knowledge_base=True,
            normalized_question=normalized,
            search_query=normalized.rstrip("？?。.!！"),
            refusal_reason=None,
            reason="问题同时包含光伏运维领域词和故障/处理动作词，强制走知识库",
        )

    return None


async def understand_query(
    question: str,
    chat_client: ChatClient,
    max_question_chars: int,
) -> QueryUnderstandingResult:
    hard_rule_result = apply_intent_hard_rules(question, max_question_chars)
    if hard_rule_result is not None:
        return hard_rule_result

    normalized = normalize_query(question)
    try:
        raw = await chat_client.chat(
            messages=[
                {"role": "system", "content": _intent_system_prompt()},
                {"role": "user", "content": normalized},
            ],
            temperature=0.0,
        )
        model_result = _parse_model_result(raw, fallback_question=normalized)
    except Exception:
        model_result = QueryUnderstandingResult(
            intent=Intent.knowledge_base_qa,
            confidence=0.0,
            should_use_knowledge_base=True,
            normalized_question=normalized,
            search_query=normalized,
            refusal_reason=None,
            reason="意图识别失败，按知识库检索兜底",
        )

    return post_validate_understanding(
        original_question=normalized,
        model_result=model_result,
    )


def post_validate_understanding(
    original_question: str,
    model_result: QueryUnderstandingResult,
) -> QueryUnderstandingResult:
    normalized_original = normalize_query(original_question)
    if _is_domain_fault_question(normalized_original) and model_result.intent != Intent.knowledge_base_qa:
        normalized_question = model_result.normalized_question or normalized_original
        return QueryUnderstandingResult(
            intent=Intent.knowledge_base_qa,
            confidence=model_result.confidence,
            should_use_knowledge_base=True,
            normalized_question=normalized_question,
            search_query=model_result.search_query or normalized_question,
            refusal_reason=None,
            reason="后处理发现领域故障处理问题，覆盖模型意图为知识库问答",
        )

    if model_result.intent == Intent.knowledge_base_qa and not model_result.search_query:
        return QueryUnderstandingResult(
            intent=model_result.intent,
            confidence=model_result.confidence,
            should_use_knowledge_base=True,
            normalized_question=model_result.normalized_question or normalized_original,
            search_query=model_result.normalized_question or normalized_original,
            refusal_reason=None,
            reason=model_result.reason,
        )

    return model_result


def _parse_model_result(raw: str, fallback_question: str) -> QueryUnderstandingResult:
    data = json.loads(_extract_json(raw))
    intent = Intent(data["intent"])
    return QueryUnderstandingResult(
        intent=intent,
        confidence=float(data.get("confidence", 0.0)),
        should_use_knowledge_base=bool(data.get("should_use_knowledge_base", intent == Intent.knowledge_base_qa)),
        normalized_question=str(data.get("normalized_question") or fallback_question),
        search_query=str(data.get("search_query") or ""),
        refusal_reason=data.get("refusal_reason"),
        reason=str(data.get("reason") or ""),
    )


def _extract_json(raw: str) -> str:
    stripped = raw.strip()
    if stripped.startswith("```"):
        stripped = re.sub(r"^```(?:json)?", "", stripped).strip()
        stripped = re.sub(r"```$", "", stripped).strip()
    start = stripped.find("{")
    end = stripped.rfind("}")
    if start == -1 or end == -1 or end < start:
        raise ValueError("No JSON object found in intent response")
    return stripped[start : end + 1]


def _intent_system_prompt() -> str:
    return """你只做意图识别和查询改写，不回答用户问题。
输出严格 JSON，不要输出 Markdown。
可选 intent:
- knowledge_base_qa: 光伏运维、设备故障、排查、处理、安全、巡检、制度等需要基于项目知识库回答的问题。
- general_explanation: 通用概念解释，可以直接用模型自身知识回答。
- out_of_scope: 与光伏运维和通用解释都无关的问题。
- realtime_external: 需要天气、新闻、股价、价格、汇率、当前时间等实时外部信息的问题。
- invalid_input: 空问题、无意义问题或无法理解的问题。
规则:
1. 不要改变用户原意。
2. 不要添加用户没有明确表达的具体故障原因。
3. 必须保留设备名、故障码、型号、英文缩写，例如 SVG、MPPT、PV、PID、10kV。
4. 属于光伏运维故障、排查、处理、安全、巡检、制度的问题必须选择 knowledge_base_qa。
5. 如果只是通用概念解释，可以选择 general_explanation。
输出字段:
{
  "intent": "...",
  "confidence": 0.0,
  "should_use_knowledge_base": true,
  "normalized_question": "...",
  "search_query": "...",
  "refusal_reason": null,
  "reason": "..."
}
"""


def _is_invalid_question(question: str, max_question_chars: int) -> bool:
    if not question:
        return True
    if len(question) > max_question_chars:
        return True
    return re.search(r"[\w\u4e00-\u9fff]", question) is None


def _contains_any(text: str, terms: tuple[str, ...]) -> bool:
    lowered = text.lower()
    return any(term.lower() in lowered for term in terms)


def _is_domain_fault_question(question: str) -> bool:
    return _contains_any(question, DOMAIN_TERMS) and _contains_any(question, FAULT_ACTION_TERMS)

```python
from typing import Protocol


class ChatClient(Protocol):
    async def chat(
        self,
        messages: list[dict[str, str]],
        temperature: float = 0.1,
    ) -> str:
        ...


async def generate_rag_answer(
    chat_client: ChatClient,
    question: str,
    evidence: list[object],
    cautious: bool,
) -> str:
    system_prompt = """你是光伏运维知识库问答助手。
只能基于给定证据回答。
如果证据不足，必须说明当前知识库依据不足。
不要编造厂家参数、故障码、设备型号或标准值。
回答面向光伏运维人员，优先给可能原因、排查步骤、处理建议和安全注意事项。
"""
    if cautious:
        system_prompt += "当前检索置信度中等，回答必须使用谨慎语气，并说明“根据当前知识库中相关片段”。\n"

    user_prompt = _build_rag_user_prompt(question, evidence)
    return await chat_client.chat(
        messages=[
            {"role": "system", "content": system_prompt},
            {"role": "user", "content": user_prompt},
        ],
        temperature=0.1,
    )


async def generate_general_answer(
    chat_client: ChatClient,
    question: str,
) -> str:
    return await chat_client.chat(
        messages=[
            {
                "role": "system",
                "content": "你是通用问答助手。本次回答不引用项目知识库。回答要简洁、准确，不要声称来自项目知识库。",
            },
            {"role": "user", "content": question},
        ],
        temperature=0.1,
    )


def _build_rag_user_prompt(question: str, evidence: list[object]) -> str:
    parts = [f"用户问题：{question}", "", "证据片段："]
    for index, item in enumerate(evidence, start=1):
        heading_path = getattr(item, "heading_path", "")
        clean_text = getattr(item, "clean_text", "")
        rerank_score = getattr(item, "rerank_score", None)
        parts.append(
            f"证据 {index}\n"
            f"标题路径：{heading_path}\n"
            f"rerank_score：{rerank_score}\n"
            f"内容：{clean_text[:1200]}"
        )
    parts.append("")
    parts.append("请基于以上证据回答用户问题。")
    return "\n\n".join(parts)
```

```python
async def answer_question(

    trace_id = uuid.uuid4().hex
    qa_session = _get_or_create_session(session, session_id)

    if understanding.intent in {Intent.invalid_input, Intent.realtime_external, Intent.out_of_scope}:

        return _response_from_record(record, understanding.intent.value, [], "refused", False)

    if understanding.intent == Intent.general_explanation and not understanding.should_use_knowledge_base:

def _get_or_create_session(session: Session, session_id: str | None) -> QaSession:
    qa_session = QaSession(id=uuid.UUID(session_id)) if session_id else QaSession()
    session.add(qa_session)
    session.flush()
    return qa_session


def _add_record(
    session: Session,
    qa_session: QaSession,
    trace_id: str,
    question: str,
    understanding: QueryUnderstandingResult,
    answer: str,
    answer_type: AnswerType,
    confidence: float | None,
    latency_ms: int,
    decision_extra: dict,
) -> QaRecord:
    metadata = {
        "intent": understanding.intent.value,
        "should_use_knowledge_base": understanding.should_use_knowledge_base,
        "normalized_question": understanding.normalized_question,
        "search_query": understanding.search_query,
        "intent_confidence": understanding.confidence,
        "intent_reason": understanding.reason,
        **decision_extra,
    }
    record = QaRecord(
        session_id=qa_session.id,
        trace_id=trace_id,
        question=question,
        normalized_question=understanding.normalized_question,
        answer=answer,
        answer_type=answer_type,
        confidence=confidence,
        model_name=None,
        latency_ms=latency_ms,
        decision_metadata=metadata,
    )
    session.add(record)
    session.flush()
    return record


def _add_references(
    session: Session,
    record: QaRecord,
    evidence: list[object],
) -> list[QaReferenceSchema]:
    schemas: list[QaReferenceSchema] = []
    for index, item in enumerate(evidence, start=1):
        reference = QaReference(
            qa_record_id=record.id,
            document_id=_uuid_or_none(getattr(item, "document_id", None)),
            segment_id=_uuid_or_none(getattr(item, "segment_id", None)),
            rank=index,
            relevance_score=getattr(item, "rerank_score", None),
            vector_score=getattr(item, "vector_score", None),
            keyword_score=getattr(item, "keyword_score", None),
            rrf_score=getattr(item, "rrf_score", None),
            excerpt=getattr(item, "clean_text", "")[:500],
            ref_metadata={"heading_path": getattr(item, "heading_path", "")},
        )
        session.add(reference)
        schemas.append(_reference_schema(index, item))
    return schemas


def _add_unanswered(
    session: Session,

            normalized_question=understanding.normalized_question,

        )
    )


def _response_from_record(
    record: QaRecord,
    intent: str,
    references: list[QaReferenceSchema],
    route: str,
    used_knowledge_base: bool,
) -> QaAskResponse:
    decision = dict(record.decision_metadata or {})
    decision.setdefault("route", route)
    decision.setdefault("used_knowledge_base", used_knowledge_base)
    return QaAskResponse(
        trace_id=record.trace_id or "",
        answer_type=record.answer_type.value,
        intent=intent,
        answer=record.answer or "",
        confidence=record.confidence,
        references=references,
        decision=decision,
    )


def _reference_schema(rank: int, item: object) -> QaReferenceSchema:
    return QaReferenceSchema(
        rank=rank,
        segment_id=str(getattr(item, "segment_id", "")) or None,
        document_id=str(getattr(item, "document_id", "")) or None,
        heading_path=getattr(item, "heading_path", "") or "",
        excerpt=(getattr(item, "clean_text", "") or "")[:500],
        vector_score=getattr(item, "vector_score", None),
        keyword_score=getattr(item, "keyword_score", None),
        rrf_score=getattr(item, "rrf_score", None),
        rerank_score=getattr(item, "rerank_score", None),
    )


def _top_rerank_score(evidence: list[object]) -> float | None:
    if not evidence:
        return None
    score = getattr(evidence[0], "rerank_score", None)
    return float(score) if score is not None else None


def _latency_ms(start: float) -> int:
    return int((time.perf_counter() - start) * 1000)


def _uuid_or_none(value: object) -> uuid.UUID | None:
    if value is None:
        return None
    if isinstance(value, uuid.UUID):
        return value
    try:
        return uuid.UUID(str(value))
    except ValueError:
        return None


def _refusal_message(understanding: QueryUnderstandingResult) -> str:
    if understanding.intent == Intent.invalid_input:
        return "请补充一个明确的问题。"
    if understanding.intent == Intent.realtime_external:
        return "当前系统未接入实时外部数据源，暂时无法回答该实时信息问题。"
    return "当前问题不在本系统支持范围内。"
```

```python
def get_qa_answerer() -> QaAnswerer:
    async def _answer(request: QaAskRequest) -> QaAskResponse:
        async with (
            httpx.AsyncClient(base_url=settings.llm_base_url, timeout=60) as llm_http,
            httpx.AsyncClient(base_url=settings.embedding_base_url, timeout=60) as embedding_http,
            httpx.AsyncClient(base_url=settings.rerank_base_url, timeout=60) as rerank_http,
        ):
            intent_chat_client = SiliconFlowChatClient(
                client=llm_http,
                api_key=settings.llm_api_key,
                model=settings.qa_intent_model,
            )
            answer_chat_client = SiliconFlowChatClient(
                client=llm_http,
                api_key=settings.llm_api_key,
                model=settings.qa_chat_model,
            )
            embedding_client = SiliconFlowEmbeddingClient(
                client=embedding_http,
                api_key=settings.embedding_api_key,
                model=settings.embedding_model,
                dimension=settings.embedding_dimension,
            )
            rerank_client = (
                SiliconFlowRerankClient(
                    client=rerank_http,

                    model=settings.rerank_model,
                )
                if settings.rerank_enabled
                else None
            )
            with SessionLocal() as session:
                return await answer_question(
                    session=session,

                    session_id=request.session_id,
```
